# Pico W LED Controller
import network
import socket
import machine
import neopixel
import time
from machine import Pin, ADC

# Initialize components
led_pin = Pin(2, Pin.OUT)
np = neopixel.NeoPixel(led_pin, 60)  # 60 LED strip
sound_sensor = ADC(Pin(26))

# WiFi connection
def connect_wifi():
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    wlan.connect('your_wifi', 'your_password')
    
    while wlan.isconnected() == False:
        print('Waiting for connection...')
        time.sleep(1)
    
    print(f'Connected to WiFi: {wlan.ifconfig()}')

# LED patterns based on sound input
def update_leds():
    sound_level = sound_sensor.read_u16()
    brightness = int((sound_level / 65535) * 255)
    
    # Create rainbow effect based on sound
    for i in range(60):
        hue = (i * 4 + int(time.ticks_ms() / 100)) % 255
        np[i] = hsv_to_rgb(hue, 255, brightness)
    
    np.write()

def hsv_to_rgb(h, s, v):
    # HSV to RGB conversion
    h = h / 255.0 * 360
    s = s / 255.0
    v = v / 255.0
    
    c = v * s
    x = c * (1 - abs((h / 60) % 2 - 1))
    m = v - c
    
    if 0 <= h < 60:
        r, g, b = c, x, 0
    elif 60 <= h < 120:
        r, g, b = x, c, 0
    elif 120 <= h < 180:
        r, g, b = 0, c, x
    elif 180 <= h < 240:
        r, g, b = 0, x, c
    elif 240 <= h < 300:
        r, g, b = x, 0, c
    else:
        r, g, b = c, 0, x
    
    return (int((r + m) * 255), int((g + m) * 255), int((b + m) * 255))

# Main loop
connect_wifi()
while True:
    update_leds()
    time.sleep(0.1)
